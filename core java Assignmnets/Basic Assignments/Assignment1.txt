 class armstrong {
	public boolean abc(int num)
	{
		int n=0,r=0;
		int c=num;
			while(num>0)
			{
				 n=num%10;
				n=n*n*n;
				r=r+n;
				num=num/10;
			}
			if(r==c)
				return true;
			else
				return false;
		}
		
		
	

}

public class assignment {

	{
		Scanner sc=new Scanner(System.in);
		int num=sc.nextInt();
		armstrong a=new armstrong();
		System.out.print(a.abc(num));
	}
